int shared_ptr_function()
{
    auto pointer1 = make_shared<int>(10);
    auto pointer2 = pointer1;
    auto pointer3 = pointer1;
    //获得原始指针
    auto* p = pointer1.get();


    //显示引用次数
    log(pointer1.use_count());
    log(pointer2.use_count());

    //reset() 减少一个引用计数

    pointer3.reset();
    log(pointer1.use_count());//1
    log(pointer2.use_count());//0
    log(pointer3.use_count());//0

    return 0;
}

std::weak_ptr

int weak_ptr_func()
{
    struct A;
    struct B;
    struct A {
        shared_ptr<B> pointer;
        ~A() {
            std::cout << "A 被销毁" << std::endl;
        }
    };
    struct B {
        shared_ptr<A> pointer;
        ~B() {
            std::cout << "B 被销毁" << std::endl;
        }
    };
    auto a = std::make_shared<A>();  //引用计数一次
    auto b = std::make_shared<B>();  //引用计数一次
    a->pointer = b;                  //引用计数两次
    b->pointer = a;                     //引用计数两次
    log(a.use_count()); //2
    log(b.use_count()); //2
    //退出作用域时 引用减少一次 则为1 无法释放内存
    //此时使用弱引用指针 weak_ptr 因为弱引用不会增加引用次数 问题得以解决

    //解决方法  二次使用弱引用 weak_ptr 
    struct C;
    struct D;
    struct C {
        weak_ptr<D> pointer;
        ~C() {
            std::cout << "A 被销毁" << std::endl;
        }
    };
    struct D {
        weak_ptr<C> pointer;
        ~D() {
            std::cout << "B 被销毁" << std::endl;
        }
    };
    auto c = std::make_shared<C>();  
    auto d = std::make_shared<D>();  
    c->pointer = d;                  
    d->pointer = c;                     
    log(c.use_count()); //1
    log(d.use_count()); //1

    //当作用域结束后 引用则为0 析构函数执行
    return 0;
}

条件变量 condition_variable

简单实现

#include <queue>
#include <chrono>
#include <mutex>
#include <thread>
#include <iostream>
#include <condition_variable>
int main(int argc, char const* argv[]) {
    //STL队列 只能从容器末尾添加 并且只能从容器头部删除  
    //生产者队列
    std::queue<int> produced_nums;
    //互斥量
    std::mutex mtx;
    //条件变量
    std::condition_variable cv;
    //状态
    bool notified = false;  // 通知信号
    // 生产者
    auto producer = [&]() {
        for (int i = 0; ; i++) {
            std::this_thread::sleep_for(std::chrono::milliseconds(900));
            std::unique_lock<std::mutex> lock(mtx);
            std::cout << "producing " << i << std::endl;
            produced_nums.push(i);
            notified = true;
            cv.notify_all(); // 此处也可以使用 notify_one
        }
    };
    // 消费者
    auto consumer = [&]() {
        while (true) {
            std::unique_lock<std::mutex> lock(mtx);
            while (!notified) {  // 避免虚假唤醒
                cv.wait(lock);
            }
            // 短暂取消锁，使得生产者有机会在消费者消费空前继续生产
            lock.unlock();
            //消费者慢于生产者
            std::this_thread::sleep_for(std::chrono::milliseconds(1000)); 
            
            lock.lock();
            while (!produced_nums.empty()) {
                std::cout << "consuming " << produced_nums.front() 
                    << "Peoduced SIZE： " <<  produced_nums.size()  << std::endl;
                produced_nums.pop();
            }
            notified = false;
        }
    };
    //一个生产者 两个消费者
    // 分别在不同的线程中运行
    std::thread p(producer);
    std::thread cs[2];
    //两个消费者
    for (int i = 0; i < 2; ++i) {
        cs[i] = std::thread(consumer);
    }
    p.join();
    for (int i = 0; i < 2; ++i) {
        cs[i].join();
    }
    return 0;
}
