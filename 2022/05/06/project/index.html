一. 固件更新

二. 使用stm32进行串口配置

简单用了一下freertos和串口dma
大致就是利用dma 串口1接收到内存
开了一个任务 中断发生就通知任务回显接收的数据
就实现了通过stm32控制esp8266
有点复杂 纯属就是为了玩一下 如果需要移植 还需要修改

esp8266引脚初始化

ESP8266_StatusTypeDef InitEsp8266(void){
    HAL_GPIO_WritePin(GPIOE, GPIO_PIN_10, GPIO_PIN_SET);  //ESP8266EN
    HAL_GPIO_WritePin(GPIOE, GPIO_PIN_11, GPIO_PIN_RESET);
    HAL_Delay(200);
    HAL_GPIO_WritePin(GPIOE, GPIO_PIN_11, GPIO_PIN_SET);  // ESP8266 reseted
    HAL_Delay(2000);
    return ESP8266_OK;
}
//一个en 一个reset 随便两个引脚即可

{
  /* USER CODE BEGIN USART1_IRQn 0 */
    temp = __HAL_UART_GET_FLAG(&huart1, UART_FLAG_IDLE);
    if(temp != RESET){
        temp = huart1.Instance->ISR;
        temp = huart1.Instance->RDR;
        HAL_UART_DMAStop(&huart1);
        temp = __HAL_DMA_GET_COUNTER(&hdma_usart1_rx);
        U1RxBufferLen = BUFFER_SIZE_MAX - temp;
        osThreadFlagsSet(DriverTaskHandle, 0x01);  // 通知 DriverTaskHandle task
        __HAL_UART_CLEAR_IDLEFLAG(&huart1);
        HAL_UART_Receive_DMA(&huart1, U1RxBuffer, BUFFER_SIZE_MAX);
    }
  /* USER CODE END USART1_IRQn 0 */

}

起始信号

void MPU6050_Start()  
{
    WriteSDA_1();
    WriteSCL_1();
    delay_us(5);
    WriteSDA_0();
    delay_us(5);
    WriteSCL_0();
    delay_us(5);
}

停止信号
**************************************/
void MPU6050_Stop()  
{    
   WriteSDA_0();
   WriteSCL_1();
   delay_us(5);
   WriteSDA_1();
}

//I2C发送应答信号
//入口参数:ack (0:ACK 1:NAK)
void MPU6050_SendACK(uchar ack)
{
    
    SDA=ack;                     //写应答信号
    WriteSCL_1();                   //拉高时钟线
    delay_us(5);//Delay5us();    //延时
    WriteSCL_0();                   //拉低时钟线
    delay_us(5);//Delay5us();    //延时
}

/**************************************
发送应答信号
**************************************/
uchar MPU6050_RecvACK()
{
    ReadSDA();
    WriteSCL_1();                       //拉高时钟线
    delay_us(5);//Delay5us();    //延时
    CY=SDA;                      //读应答信号
    WriteSCL_0();                       //拉低时钟线
    delay_us(5);//Delay5us();    //延时
    P0DIR|=0x00;
    return CY;
}

/**************************************
向IIC总线发送一个字节数据
**************************************/
void MPU6050_SendByte(uchar data)
{
    uchar i;
    SDA_OUT();
    
    for(i=0;i<8;i++)         //8位计数器
    {       
        if(data&0x80)
        {WriteSDA_1();}
        else 
        {WriteSDA_0();}
        
        data<<=1;
        WriteSCL_1();            //拉高时钟线
        delay_us(5);          //延时
        WriteSCL_0();            //拉低时钟线
        delay_us(5);          //延时
    }
    MPU6050_RecvACK();
}

/**************************************
从IIC总线接收一个字节数据
**************************************/
uchar MPU6050_RecvByte()
{
  
    uchar i;
    uchar dat = 0;
    WriteSDA_1();                  //使能内部上拉,准备读取数据,
    for (i=0; i<8; i++)         //8位计数器
    {
        dat <<= 1;
        SCL = 1;                //拉高时钟线
        delay_us(5);                 //延时
        dat |= SDA;             //读数据               
        SCL = 0;                //拉低时钟线
         delay_us(5);                 //延时
    }
    return dat;

}

//******单字节写入*******************************************
void Single_Write_MPU6050(uchar REG_Address,uchar REG_data)
{
    MPU6050_Start();                  //起始信号
    MPU6050_SendByte(SlaveAddress);   //发送设备地址+写信号
    MPU6050_SendByte(REG_Address);    //内部寄存器地址，
    MPU6050_SendByte(REG_data);       //内部寄存器数据，
    MPU6050_Stop();                   //发送停止信号
}

//********单字节读取*****************************************
uchar Single_Read_MPU6050(uchar REG_Address)
{
    uchar REG_data;
    MPU6050_Start();                   //起始信号
    MPU6050_SendByte(SlaveAddress);    //发送设备地址+写信号
    MPU6050_SendByte(REG_Address);     //发送存储单元地址，从0开始    
    MPU6050_Start();                   //起始信号
    MPU6050_SendByte(SlaveAddress+1);  //发送设备地址+读信号
    REG_data=MPU6050_RecvByte();       //读出寄存器数据
    MPU6050_SendACK(1);                //接收应答信号
    MPU6050_Stop();                    //停止信号
    return REG_data;
}

姿态解算方法

计算正确的三轴加速度和角速度值

角速度参数为：2000deg/s
加速度参数为：2g
加速度灵敏度为 ((2^16)/2) / 2 = 16384
角速度灵敏度为 ((2^16)/2) / 2000 = 16.4
例如: ED2: +03284 +07958 +12160 -00082 -00038 +00016
+03284 +07958 +12160 为 gx gy gz
-00082 -00038 +00016 为 ax ay az
则真实gx为：
gx = 3284 / 16384 = 0.2004….. （gy gz同理）
真实ax为：
ax = 82 / 16.4 = 5 (ay az同理)

计算欧拉角

根据三轴加速度和角速度得出欧拉角。 

import math
import csv
import serial
import time
class get_Euler_from_serial(object):
    def __init__(self, portx, bps, timeout):
        self.ser = serial.Serial(portx, bps, timeout = timeout) 
        self.ser.flushInput() 
        self.acceleration_Sensitivity = 16384  ## 加速度灵敏度2g
        self.Gyroscope_Sensitivity = 16.4      ## 加速度灵敏度 2000reg/s
        self.gx, self.gy, self.gz, self.ax, self.ay, self.az = 0, 0, 0, 0, 0, 0
        self.pitch, self.roll, self.yaw = 0, 0, 0

    def get_Euler(self, str):
        self.gx = float(str[6:12]) / self.acceleration_Sensitivity
        self.gy = float(str[13:20]) / self.acceleration_Sensitivity
        self.gz = float(str[21:28]) / self.acceleration_Sensitivity
        self.ax = float(str[29:36]) / self.Gyroscope_Sensitivity
        self.ay = float(str[37:44]) / self.Gyroscope_Sensitivity
        self.az = float(str[45:52]) / self.Gyroscope_Sensitivity
        Kp=100        #比例增益控制加速度计/磁强计的收敛速度
        Ki=0.002      #积分增益控制陀螺仪偏差的收敛速度
        halft=0.001   #采样周期的一半
        #传感器框架相对于辅助框架的四元数（初始化四元数的值）
        q0=1
        q1=0
        q2=0
        q3=0
        #由Ki缩放的积分误差项（初始化）
        exInt=0
        eyInt=0
        ezInt=0
        aax = self.gx
        aay = self.gy
        aaz = self.gz
        ggx = self.ax
        ggy = self.ay
        ggz = self.az
        ax = float(aax)
        ay = float(aay)
        az = float(aaz)
        gx = float(ggx)
        gy = float(ggy)
        gz = float(ggz)
        #单元化
        norm=math.sqrt(ax*ax+ay*ay+az*az)
        ax=ax/norm
        ay=ay/norm
        az=az/norm
        #估计方向的重力
        vx=2*(q1*q3-q0*q2)
        vy=2*(q0*q1+q2*q3)
        vz=q0*q0-q1*q1-q2*q2+q3*q3
        # 错误的领域和方向传感器测量参考方向之间的交叉乘积的总和
        ex=(ay*vz - az*vy)
        ey=(az*vx - ax*vz)
        ez=(ax*vy - ay*vx)
        # 积分误差比例积分增益
        exInt +=ex*Ki
        eyInt +=ey*Ki
        ezInt +=ez*Ki
        # 调整后的陀螺仪测量
        gx +=Kp*ex + exInt
        gy +=Kp*ey + eyInt
        gz +=Kp*ez +ezInt
        # 整合四元数
        q0 +=(-q1*gx - q2*gy - q3*gz)*halft
        q1 +=(q0*gx + q2*gz - q3*gy)*halft
        q2 +=(q0*gy - q1*gz + q3*gx)*halft
        q3 +=(q0*gz + q1*gy - q2*gx)*halft
        # 正常化四元数
        norm = math.sqrt(q0*q0+q1*q1+q2*q2+q3*q3)
        q0 /=norm
        q1 /=norm
        q2 /=norm
        q3 /=norm
        # 获取欧拉角 pitch、roll、yaw
        pitch = math.asin(-2*q1*q3+2*q0*q2)*57.3
        roll = math.atan2(2*q2*q3+2*q0*q1,-2*q1*q1-2*q2*q2+1)*57.3
        yaw = math.atan2(2*(q1*q2 + q0*q3),q0*q0+q1*q1-q2*q2-q3*q3)*57.3
        #打印姿态角信息以及把姿态角数据写到txt文本中
        if str[0:3] == "ED1" :
                print("ED1: Pitch:%.3f,Roll: %.3f,Yaw:%.3f" %(pitch,roll,yaw))
        if str[0:3] == "ED2" :
                print("ED2: Pitch:%.3f,Roll: %.3f,Yaw:%.3f" %(pitch,roll,yaw))

    def run(self):
        while True:
            count = self.ser.inWaiting() # 获取串口缓冲区数据
            if count !=0 :
                recv = self.ser.read(self.ser.in_waiting).decode("gbk") # 读出串口数据，数据采用gbk编码
                self.get_Euler(recv)
            time.sleep(0.2)

if __name__ == '__main__':
    get_Euler_from_serial("COM10", 115200, 5).run()
